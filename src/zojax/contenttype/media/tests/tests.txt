=====
Video
=====

    >>> from zope import component, interface
    >>> from zope.testbrowser.testing import Browser
    >>> from zojax.folder.folder import Folder

    >>> root = getRootFolder()
    >>> root['folder'] = Folder('Folder')

    >>> actor = Browser()
    >>> actor.handleErrors = False
    >>> actor.addHeader("Authorization", "Basic mgr:mgrpw")

Install product

    >>> actor.open("http://localhost/settings/")
    >>> actor.getLink('Products management').click()
    >>> actor.getControl(name='availproducts:list').value = ['zojax-contenttypes']
    >>> actor.getControl(name='install').click()

Add video

    >>> actor.open("http://localhost/folder/context.html")
    >>> actor.getLink('Video').click()

    >>> actor.getControl('Title').value = u'Test video'
    >>> actor.getControl('Video service').value = u'YouTube'
    >>> actor.getControl(name="form.widgets.video").value = u'<object>test</object>'
    >>> actor.getControl('Body').value = u'Page body'
    >>> actor.getControl(name='add_input_name').value = 'test-video'
    >>> actor.getControl(name='form.buttons.add').click()

View

    >>> print actor.contents
    <!DOCTYPE html PUBLIC
    ...

    <div class="z-content-body">

Edit

    >>> actor.open("http://localhost/folder/test-video/context.html")
    >>> actor.getControl('Title').value = 'New video'
    >>> actor.getControl('Text position').value = [u'2']
    >>> actor.getControl(name='content.edit.buttons.save').click()

    >>> actor.open("http://localhost/folder/test-video/")
    >>> print actor.contents

    </html>

Unistall product

    >>> actor.open("http://localhost/settings/")
    >>> actor.getLink('Products management').click()
    >>> actor.getControl(name='products:list').value = ['zojax-contenttypes']
    >>> actor.getControl(name='uninstall').click()
